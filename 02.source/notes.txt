notas - agencias bancarias

parâmetros de entrada:
	ghr: graus hora de resfriamento em graus celsius
	log_ghr: ghr 'normalizado'
	area: area da tipologia em m2 [fixa]
	total_cons: consumo total anual em kwh/ano
	total_cons_area: consumo total anual em kwh/m2/ano
	atm_light_cons: consumo anual da iluminacao area de atm em kwh/ano
	atm_equip_cons: consumo anual dos caixas eletronicos da area de atm em kwh/ano [fixo - nao variam os numeros de caixas eletronicos]
	atm_hvac_cons: consumo anual da area de atm com refrigeracao em kwh/ano
	atm_cons: atm_light_cons + atm_equip_cons + atm_hvac_cons: consumo anual da area de atm em kwh/ano
	hvac_equip_cons: consumo total anual com refrigeração e equipamentos em kwh/ano
	equip_cons: consumo total anual com equipamentos

parâmetros de saída:
	total end uses - consumo total final (kwh/m2/ano)
	zona atm - consumo da zona de atm
	consumo 1 - total - atm
	consumo 2 - total - atm - equipamentos
